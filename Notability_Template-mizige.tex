% ====== MIZIGE GRID SETTINGS (numeric inches) ======
% Size of each character box (typical ranges: 0.4–0.6in)
\newcommand{\BoxSizeIn}{0.50}

% Start the grid below the header area (increase to push grid down)
\newcommand{\GridTopOffsetIn}{0.75}

% Line styling
\newcommand{\OuterLineWidth}{0.30pt}
\newcommand{\InnerLineWidth}{0.20pt}
\newcommand{\OuterOpacity}{0.30}
\newcommand{\InnerOpacity}{0.22}

% ====== BACKGROUND: MIZIGE GRID ======
\AddToShipoutPictureBG{%
  \AtPageLowerLeft{%
    \begin{tikzpicture}[remember picture,overlay]
      % Work in inches with origin at page bottom-left
      \begin{scope}[shift={(current page.south west)}, x=1in, y=1in]

        % US Letter size in inches
        \pgfmathsetmacro{\PageW}{8.5}
        \pgfmathsetmacro{\PageH}{11.0}

        % Grid region top (below header)
        \pgfmathsetmacro{\YTop}{\PageH - \GridTopOffsetIn}

        % Box geometry
        \pgfmathsetmacro{\s}{\BoxSizeIn}

        % Clip so nothing draws in the header band
        \begin{scope}
          \clip (0,0) rectangle (\PageW,\YTop);

          % Overshoot so edges fill cleanly
          \pgfmathsetmacro{\XMin}{-1.5*\s}
          \pgfmathsetmacro{\XMax}{\PageW + 1.5*\s}
          \pgfmathsetmacro{\YMin}{-1.5*\s}
          \pgfmathsetmacro{\YMax}{\YTop + 1.5*\s}

          % How many boxes to draw
          \pgfmathtruncatemacro{\NCols}{ceil((\XMax-\XMin)/\s) + 1}
          \pgfmathtruncatemacro{\NRows}{ceil((\YMax-\YMin)/\s) + 1}

          % Draw repeating Mizige boxes:
          % outer square + center vertical/horizontal + both diagonals
          \foreach \r in {0,...,\NRows} {
            \pgfmathsetmacro{\y}{\YMin + \r*\s}
            \foreach \c in {0,...,\NCols} {
              \pgfmathsetmacro{\x}{\XMin + \c*\s}

              % Outer square
              \draw[line width=\OuterLineWidth, opacity=\OuterOpacity]
                (\x,\y) rectangle ++(\s,\s);

              % Inner guidelines (the "米" structure)
              \draw[line width=\InnerLineWidth, opacity=\InnerOpacity]
                (\x,\y) -- ++(\s,\s);                % diagonal \
              \draw[line width=\InnerLineWidth, opacity=\InnerOpacity]
                (\x,\y+\s) -- ++(\s,-\s);             % diagonal /
              \draw[line width=\InnerLineWidth, opacity=\InnerOpacity]
                (\x+\s/2,\y) -- ++(0,\s);             % vertical center
              \draw[line width=\InnerLineWidth, opacity=\InnerOpacity]
                (\x,\y+\s/2) -- ++(\s,0);             % horizontal center
            }
          }
        \end{scope}
      \end{scope}
    \end{tikzpicture}%
  }%
}
