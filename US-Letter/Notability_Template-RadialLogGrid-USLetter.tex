% ===== GRID SETTINGS =====
\newcommand{\GridTopOffsetIn}{1.00}
\newcommand{\SpokeStepDeg}{10}

% Smallest base decade radius
\newcommand{\BaseRadiusIn}{0.10}

% Styling
\newcommand{\MinorWidth}{0.2pt}
\newcommand{\MidWidth}{0.35pt}
\newcommand{\MajorWidth}{0.6pt}

\newcommand{\MinorOpacity}{0.1}
\newcommand{\MidOpacity}{0.15}
\newcommand{\MajorOpacity}{0.2}

% ===== BACKGROUND =====
\AddToShipoutPictureBG{%
\AtPageLowerLeft{%
\begin{tikzpicture}[remember picture,overlay]
\begin{scope}[shift={(current page.south west)}, x=1in, y=1in]

\def\PageW{8.5}
\def\PageH{11}

\pgfmathsetmacro{\YTop}{\PageH-\GridTopOffsetIn}
\clip (0,0) rectangle (\PageW,\YTop);

\pgfmathsetmacro{\Cx}{\PageW/2}
\pgfmathsetmacro{\Cy}{\YTop/2}
\pgfmathsetmacro{\Rmax}{sqrt((\PageW/2)^2 + (\YTop/2)^2) + 0.2}

\pgfmathsetmacro{\Rbase}{\BaseRadiusIn}

% ---- Determine number of decades needed ----
\pgfmathsetmacro{\MaxDecade}{ceil(log10(\Rmax/\Rbase))}

% ---- Draw Log Rings (1–9 × 10^n) ----
\foreach \d in {0,...,\MaxDecade} {

  \foreach \m in {1,...,9} {

    \pgfmathsetmacro{\Radius}{\m * \Rbase * pow(10,\d)}

    \ifdim \Radius pt < \Rmax pt

      % Style selection
      \ifnum\m=1
        \draw[line width=\MajorWidth, opacity=\MajorOpacity]
          (\Cx,\Cy) circle[radius=\Radius];
      \else
        \ifnum\m=2
          \draw[line width=\MidWidth, opacity=\MidOpacity]
            (\Cx,\Cy) circle[radius=\Radius];
        \else
          \ifnum\m=5
            \draw[line width=\MidWidth, opacity=\MidOpacity]
              (\Cx,\Cy) circle[radius=\Radius];
          \else
            \draw[line width=\MinorWidth, opacity=\MinorOpacity]
              (\Cx,\Cy) circle[radius=\Radius];
          \fi
        \fi
      \fi

    \fi
  }
}

% ---- Spokes ----
\pgfmathtruncatemacro{\NSpokes}{ceil(360/\SpokeStepDeg)-1}
\foreach \k in {0,...,\NSpokes} {
  \pgfmathsetmacro{\ang}{\k*\SpokeStepDeg}
  \draw[line width=\MinorWidth, opacity=\MinorOpacity]
    (\Cx,\Cy) -- ({\Cx + \Rmax*cos(\ang)},
                  {\Cy + \Rmax*sin(\ang)});
}

\end{scope}
\end{tikzpicture}%
}}
