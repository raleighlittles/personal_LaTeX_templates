\newcommand{\MajorOpacity}{0.30}
\newcommand{\OuterOpacity}{0.50}

% ====== BACKGROUND: SMITH CHART ======
\AddToShipoutPictureBG{%
  \AtPageLowerLeft{%
    \begin{tikzpicture}[remember picture,overlay]
      \begin{scope}[shift={(current page.south west)}, x=1in, y=1in]

        \def\PageW{8.5}
        \def\PageH{11}
        \pgfmathsetmacro{\YTop}{\PageH - \GridTopOffsetIn}

        % Center
        \pgfmathsetmacro{\Cx}{\PageW/2}
        \pgfmathsetmacro{\Cy}{\YTop/2}

        % Oversized radius
        \pgfmathsetmacro{\ChartR}{max(\PageW,\YTop)*0.6}

        % Clip only vertically (so it doesn't hit header)
        \clip (0,0) rectangle (\PageW,\YTop);

        \begin{scope}[shift={({\Cx},{\Cy})}, scale=\ChartR]

          % Outer circle (extends off page)
          \draw[line width=\OuterLineWidth, opacity=\OuterOpacity] (0,0) circle (1);

          % Real axis
          \draw[line width=\MajorLineWidth, opacity=\MajorOpacity] (-1,0) -- (1,0);

          \begin{scope}
            \clip (0,0) circle (1);

            % Resistance circles
            \foreach \rr in {0.1,0.2,0.3,0.5,1,2,5} {
              \draw[line width=\MinorLineWidth, opacity=\MinorOpacity]
                ({\rr/(1+\rr)},0) circle ({1/(1+\rr)});
            }

            % Reactance circles
            \foreach \xx in {0.2,0.5,1,2,5} {
              \draw[line width=\MinorLineWidth, opacity=\MinorOpacity]
                (1,{1/\xx}) circle ({1/abs(\xx)});
              \draw[line width=\MinorLineWidth, opacity=\MinorOpacity]
                (1,{-1/\xx}) circle ({1/abs(\xx)});
            }

          \end{scope}

        \end{scope}
      \end{scope}
    \end{tikzpicture}%
  }%
}
