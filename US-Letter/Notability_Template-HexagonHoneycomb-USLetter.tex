% ====== HEX GRID SETTINGS (in inches, numeric) ======
% Hex side length: smaller = denser grid (try 0.10 to 0.14)
\newcommand{\HexSideIn}{0.12}

% Start the grid below the header area (in inches, numeric)
\newcommand{\GridTopOffsetIn}{0.8}

% Line appearance
\newcommand{\HexLineWidth}{0.25pt}
\newcommand{\HexOpacity}{0.10}

% ====== HEX GRID BACKGROUND ======
\AddToShipoutPictureBG{%
  \AtPageLowerLeft{%
    \begin{tikzpicture}[remember picture,overlay]
      % Work in inches with origin at page bottom-left
      \begin{scope}[shift={(current page.south west)}, x=1in, y=1in]

        % Page dimensions in inches (US Letter)
        \pgfmathsetmacro{\PageW}{8.5}
        \pgfmathsetmacro{\PageH}{11.0}

        % Grid region top (below header)
        \pgfmathsetmacro{\YTop}{\PageH - \GridTopOffsetIn}

        % Hex geometry (pointy-top hexes)
        \pgfmathsetmacro{\a}{\HexSideIn}
        \pgfmathsetmacro{\dx}{sqrt(3)*\a}   % horizontal center-to-center
        \pgfmathsetmacro{\dy}{1.5*\a}       % vertical center-to-center
        \pgfmathsetmacro{\hx}{sqrt(3)/2*\a} % x offset to slanted vertices

        % Clip so no hex lines appear in the header band
        \begin{scope}
          \clip (0,0) rectangle (\PageW,\YTop);

          % Overshoot a bit so edges fill cleanly
          \pgfmathsetmacro{\XMin}{-2*\dx}
          \pgfmathsetmacro{\XMax}{\PageW + 2*\dx}
          \pgfmathsetmacro{\YMin}{-2*\dy}
          \pgfmathsetmacro{\YMax}{\YTop + 2*\dy}

          % Draw the honeycomb: alternate row offset of dx/2
          \pgfmathtruncatemacro{\NRows}{ceil((\YMax-\YMin)/\dy) + 2}
          \pgfmathtruncatemacro{\NCols}{ceil((\XMax-\XMin)/\dx) + 2}

          \foreach \r in {0,...,\NRows} {
            \pgfmathsetmacro{\cy}{\YMin + \r*\dy}
            \pgfmathsetmacro{\xoff}{mod(\r,2)*(\dx/2)}

            \foreach \c in {0,...,\NCols} {
              \pgfmathsetmacro{\cx}{\XMin + \c*\dx + \xoff}

              % Hex outline (pointy-top)
              \draw[line width=\HexLineWidth, opacity=\HexOpacity]
                (\cx,      \cy+\a) --
                (\cx+\hx,  \cy+\a/2) --
                (\cx+\hx,  \cy-\a/2) --
                (\cx,      \cy-\a) --
                (\cx-\hx,  \cy-\a/2) --
                (\cx-\hx,  \cy+\a/2) -- cycle;
            }
          }
        \end{scope}
      \end{scope}
    \end{tikzpicture}%
  }%
}
