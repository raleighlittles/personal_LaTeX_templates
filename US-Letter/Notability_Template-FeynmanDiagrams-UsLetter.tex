\newcommand{\ShufflePermutation}{%
  \InitPermutation
  \pgfmathsetseed{\DiagramSeed}
  \foreach \i in {8,7,6,5,4,3,2} {
    \pgfmathtruncatemacro{\j}{1 + int(rnd*\i)}
    \expandafter\SwapPerms\expandafter{\i}{\j}
  }
}
