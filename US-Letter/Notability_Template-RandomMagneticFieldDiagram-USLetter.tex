\documentclass[10pt]{article}

\usepackage{tikz}

% ====== USER FIELDS ======
\newcommand{\StudentField}{Raleigh Littles}
\newcommand{\CourseField}{Class Name}
\newcommand{\DateField}{\today}
\newcommand{\TitleField}{Note title} % leave empty for title

% ====== SYMBOL GRID SETTINGS ======
\newcommand{\GridTopOffsetIn}{0.4}

% Symbol spacing
\newcommand{\SymbolXStep}{0.35}  % horizontal spacing in inches
\newcommand{\SymbolYStep}{0.35}  % vertical spacing in inches

% Symbol appearance
\newcommand{\SymbolOpacity}{0.15}
\newcommand{\SymbolScale}{2}

% Optional jitter (set to 0 for perfectly rigid placement)
\newcommand{\JitterAmount}{0.00}

% ====== BACKGROUND: MAGNETIC FIELD SYMBOL GRID ======
\AddToShipoutPictureBG{%
  \AtPageLowerLeft{%
    \begin{tikzpicture}[remember picture,overlay]
      \begin{scope}[shift={(current page.south west)}, x=1in, y=1in]

        % Page size in inches
        \def\PageW{8.5}
        \def\PageH{11.0}
        \pgfmathsetmacro{\YTop}{\PageH - \GridTopOffsetIn}

        % Clip below header
        \clip (0,0) rectangle (\PageW,\YTop);

        % Random seed (change this number if you want a different pattern)
        \pgfmathsetseed{0123456}

        % Number of columns/rows
        \pgfmathtruncatemacro{\NCols}{ceil(\PageW/\SymbolXStep)+1}
        \pgfmathtruncatemacro{\NRows}{ceil(\YTop/\SymbolYStep)+1}

        % Draw evenly spaced random mix of \otimes and \odot
        \foreach \r in {0,...,\NRows} {
          \foreach \c in {0,...,\NCols} {
          
            \newcommand{\GridXMargin}{0.20} % left/right margin in inches
            \newcommand{\GridYMargin}{0.20} % bottom margin (optional)

            \pgfmathsetmacro{\xbase}{\GridXMargin + \c*\SymbolXStep}
            \pgfmathsetmacro{\ybase}{\GridYMargin + \r*\SymbolYStep}

            % Optional slight jitter
            \pgfmathsetmacro{\jx}{(2*rnd-1)*\JitterAmount}
            \pgfmathsetmacro{\jy}{(2*rnd-1)*\JitterAmount}

            % Random symbol choice
            \pgfmathparse{rnd < 0.5 ? 1 : 0}
            \let\IntoPage\pgfmathresult

            \ifnum\IntoPage=1
              \node[opacity=\SymbolOpacity, scale=\SymbolScale]
                at ({\xbase+\jx},{\ybase+\jy}) {$\otimes$};
            \else
              \node[opacity=\SymbolOpacity, scale=\SymbolScale]
                at ({\xbase+\jx},{\ybase+\jy}) {$\odot$};
            \fi
          }
        }

      \end{scope}
    \end{tikzpicture}%
  }%
}
