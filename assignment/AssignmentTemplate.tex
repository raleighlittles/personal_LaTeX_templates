% Simple Latex template for assignments, notes, or scratch work. 
% Header includes fields for name, ID, class, date, and page number. 
% Footer includes title, name, date, and page number. 
%The header and footer are separated from the main body by horizontal lines. 
% The main body has a large section title at the top left. 
% The template uses the "fancyhdr" package for header and footer customization, and the "geometry" package for page layout. 
% The "amsmath" package is included for math formatting, and the "graphicx" package is included for adding images. The "soul" package is included for highlighting text with \hl{}.
% Copyright: Raleigh Littles <raleighlittles@arizona.edu>, 2026

\documentclass[10pt]{article}

% --------------------------------------------------
% Page geometry
% --------------------------------------------------
\usepackage[
  letterpaper,
  margin=0.10in,
  bottom=0.20in,
  headheight=48pt,
  headsep=0.4in,
  footskip=4in,
  includehead,
  includefoot
]{geometry}

% --------------------------------------------------
% Fonts & encoding
% --------------------------------------------------
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{lastpage}
\usepackage{lmodern}

% --------------------------------------------------
% Math packages (common set)
% --------------------------------------------------
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{amsthm}
\usepackage{mathtools}
\usepackage{physics}
\usepackage{siunitx}

% --------------------------------------------------
% Graphics & diagrams
% --------------------------------------------------
\usepackage{graphicx}
\usepackage{tikz}
\usepackage{circuitikz}
\usepackage{xcolor}
\usepackage{soul} % for highlighting with \hl{}

% --------------------------------------------------
% Header / Footer
% --------------------------------------------------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{} % clear defaults

% --------------------------------------------------
% User-defined metadata (EDIT THESE)
% --------------------------------------------------
\newcommand{\NoteTitle}{Note Title}
\newcommand{\StudentName}{Raleigh Littles}
\newcommand{\StudentID}{\#0123456789}
\newcommand{\ClassName}{Course Title}
\newcommand{\NoteDate}{\today}

% --------------------------------------------------
% Header layout
% --------------------------------------------------
\fancyhead[L]{%
  \Huge{\NoteTitle}
}

\fancyhead[R]{%
  \begin{tabular}{r}
    \StudentName \\
    ID: \StudentID \\
    \ClassName \\
    \NoteDate \\
    Page \thepage \hspace{1pt} of \hspace{1pt} \pageref{LastPage}
  \end{tabular}
}

% --------------------------------------------------
% Footer layout
% --------------------------------------------------
\fancyfoot[L]{%
  \textbf{\NoteTitle} --- \StudentName
}

\fancyfoot[C]{%
  Page \thepage \hspace{1pt} of \hspace{1pt} \pageref{LastPage}
  \hspace{20pt}---\hspace{20pt}
  \NoteDate
}

% If you don't want to include an image in the footer, delete this entire "fancyfoot" section below!
\fancyfoot[R]{%
  % Reserved space for a 300x300 px image (~3.1in x 3.1in)
  \begin{minipage}[c][1in][c]{1in}
    %\centering
    \includegraphics[width=0.75in,height=0.75in,keepaspectratio]{AssignmentTemplate-footer.png}
    \vspace{15pt}
  \end{minipage}
}
%%% Delete up to this line if you don't want an image in the footer

% --------------------------------------------------
% Rule lines
% --------------------------------------------------
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}

% --------------------------------------------------
% Document
% --------------------------------------------------
\begin{document}

\section{Your section name here}

% Tips:

% Use \clearpage to force figures to be on the next page, if they don't fit on the current page.

% For numbered equations, use the "equation" environment
%
% For unnumbered equations, use \[ ... \] or the "equation*" environment (requires amsmath)
%
% Use fbox for boxes around text, e.g. \fbox{ ... } or for math \boxed{ ... } (requires amsmath)
% For colored boxes, define a custom colorbox:
% \fcolorbox{red}{white}{Your text here}
% Arguments: 1) border color, 2) background color, 3) content
%
% Highlight text using \hl{ ... } (requires soul package)


\pagebreak

\end{document}
